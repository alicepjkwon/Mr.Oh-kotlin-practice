interface Runnable {
    fun run()
    fun fastRun() = println("Run Fast!!")

}

class Human : Runnable{
    var name = "KPJ"
    override fun run(){
        println("${name} is running")
    }

    override fun fastRun() {
        println("Run Fast!! ${name}")
    }


    }


val human = Human()


human.run()

human.fastRun()
